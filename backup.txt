#include <Arduino.h>
#include <SoftwareSerial.h>

// =====================================================
// UART NHAN MPU TU NANO
// UNO D6 RX <- Nano D6 TX
// UNO D7 TX -> Nano D5 RX
// =====================================================
#define MPU_RX 6
#define MPU_TX 7
SoftwareSerial MpuSerial(MPU_RX, MPU_TX);

// =====================================================
// UART GUI XUONG STM / HOVERBOARD
// UNO D4 RX <- TX STM / Hoverboard
// UNO D5 TX -> RX STM / Hoverboard
// =====================================================
#define HOVER_RX 4
#define HOVER_TX 5
SoftwareSerial HoverSerial(HOVER_RX, HOVER_TX);

// =====================================================
// LINE SENSOR
// OUT2 -> A1
// OUT3 -> A2
// OUT4 -> A3
//
// Co line       -> OUT = 0
// Khong co line -> OUT = 1
// =====================================================
#define OUT2_PIN A1
#define OUT3_PIN A2
#define OUT4_PIN A3
#define LINE_THRESHOLD 500

// =====================================================
// BUTTON
// 1 chan nut -> D2
// 1 chan nut -> GND
// =====================================================
#define BUTTON_PIN 2

// =====================================================
// HOVERBOARD / STM
// =====================================================
#define HOVER_BAUD 115200
#define START_FRAME 0xABCD

typedef struct {
  uint16_t start;
  int16_t steer;
  int16_t speed;
  uint16_t checksum;
} SerialCommand;

SerialCommand Command;

// =====================================================
// SPEED CONFIG
// =====================================================
int BASE_SPEED = 60;
int LINE_TURN_SPEED = 60;

// Toc do quay tai cho
int TURN_MAX_WHEEL_SPEED = 110;
int TURN_MIN_WHEEL_SPEED = 70;

// PID quay 90 do
float TURN_TOLERANCE = 1.0;
float TURN_EXTRA_DEGREE = 6.0;

// Quay 180 dung rieng, khong de dung 180/186 de tranh dao dau
float TURN_BACK_TARGET = 178.0;

// Neu quay trai/phai bi nguoc thi doi thanh -1
int TURN_DIRECTION_SIGN = 1;

// Neu do line bi nguoc huong thi doi thanh -1
int LINE_DIRECTION_SIGN = 1;

// PID quay goc
float TURN_KP = 1.55;
float TURN_KD = 0.35;

// =====================================================
// MPU DATA
// Nano gui: x0y0z90h
// =====================================================
String mpuBuffer = "";

int mpuX = 0;
int mpuY = 0;
int mpuZ = 0;

float currentYaw = 0;
float returnZeroYaw = 0;

unsigned long lastMPUTime = 0;

// =====================================================
// LINE DATA
// =====================================================
int raw2 = 0;
int raw3 = 0;
int raw4 = 0;

int out2 = 1;
int out3 = 1;
int out4 = 1;

int lineState = 0;
bool lineSensorEnabled = true;

// =====================================================
// DEBUG
// =====================================================
unsigned long lastDebug = 0;
const unsigned long debugInterval = 100;

// =====================================================
// ROUTE COMMAND
// VD: aThLn
// =====================================================
#define MAX_CMD 30

char routeCmd[MAX_CMD];
int routeLen = 0;

char returnCmd[MAX_CMD];
int returnLen = 0;

bool hasRoute = false;
bool robotBusy = false;
bool waitingReturn = false;

String inputBuffer = "";
bool receivingCommand = false;

// =====================================================
// MOTOR
// =====================================================
void sendHover(int16_t steer, int16_t speed)
{
  Command.start = START_FRAME;
  Command.steer = steer;
  Command.speed = speed;
  Command.checksum = Command.start ^ Command.steer ^ Command.speed;

  HoverSerial.write((uint8_t *)&Command, sizeof(Command));
}

void motorStop()
{
  sendHover(0, 0);
}

void driveWheels(int leftSpeed, int rightSpeed)
{
  int speed = (leftSpeed + rightSpeed) / 2;
  int steer = (leftSpeed - rightSpeed) / 2;

  sendHover(steer, speed);
}

// =====================================================
// QUAY TAI CHO
// dir = 1  : quay phai
// dir = -1 : quay trai
// =====================================================
void rotateInPlace(int dir, int wheelSpeed)
{
  dir = dir * TURN_DIRECTION_SIGN;

  if (dir > 0)
  {
    // Quay phai: banh trai tien, banh phai lui
    driveWheels(wheelSpeed, -wheelSpeed);
  }
  else
  {
    // Quay trai: banh trai lui, banh phai tien
    driveWheels(-wheelSpeed, wheelSpeed);
  }
}

void sendStatus(const char *msg)
{
  Serial.println(msg);
}

// =====================================================
// LINE ON / OFF
// =====================================================
void forceLineOff()
{
  lineSensorEnabled = false;

  raw2 = 0;
  raw3 = 0;
  raw4 = 0;

  out2 = 1;
  out3 = 1;
  out4 = 1;

  lineState = 0;

  Serial.println("LINE SENSOR OFF");
}

void forceLineOn()
{
  lineSensorEnabled = true;
  Serial.println("LINE SENSOR ON");
}

// =====================================================
// PRINT ROUTE
// =====================================================
void printRouteString(const char *label, char *cmdList, int len)
{
  Serial.print(label);
  Serial.print(": a");

  for (int i = 0; i < len; i++)
  {
    Serial.print(cmdList[i]);

    if (i < len - 1)
    {
      Serial.print("h");
    }
  }

  Serial.println("n");
}

// =====================================================
// RESET ALL
// =====================================================
void resetAllCommandData()
{
  routeLen = 0;
  returnLen = 0;
  hasRoute = false;
  waitingReturn = false;
  receivingCommand = false;
  inputBuffer = "";
  mpuBuffer = "";

  raw2 = 0;
  raw3 = 0;
  raw4 = 0;

  out2 = 1;
  out3 = 1;
  out4 = 1;
  lineState = 0;

  returnZeroYaw = 0;

  forceLineOn();

  for (int i = 0; i < MAX_CMD; i++)
  {
    routeCmd[i] = 0;
    returnCmd[i] = 0;
  }

  motorStop();

  Serial.println("RESET DONE - READY FOR NEW COMMAND");
}

void resetRuntimeBeforeReturn()
{
  receivingCommand = false;
  inputBuffer = "";
  mpuBuffer = "";

  raw2 = 0;
  raw3 = 0;
  raw4 = 0;

  out2 = 1;
  out3 = 1;
  out4 = 1;
  lineState = 0;

  motorStop();

  Serial.println("RESET RUNTIME BEFORE RETURN");
}

// =====================================================
// MPU
// =====================================================
void readMPU()
{
  MpuSerial.listen();

  while (MpuSerial.available())
  {
    char c = MpuSerial.read();

    if (c == 'h')
    {
      int ix = mpuBuffer.indexOf('x');
      int iy = mpuBuffer.indexOf('y');
      int iz = mpuBuffer.indexOf('z');

      if (ix != -1 && iy != -1 && iz != -1)
      {
        String sx = mpuBuffer.substring(ix + 1, iy);
        String sy = mpuBuffer.substring(iy + 1, iz);
        String sz = mpuBuffer.substring(iz + 1);

        mpuX = sx.toInt();
        mpuY = sy.toInt();
        mpuZ = sz.toInt();

        currentYaw = mpuZ;
        lastMPUTime = millis();
      }

      mpuBuffer = "";
    }
    else if (c != '\n' && c != '\r')
    {
      mpuBuffer += c;

      if (mpuBuffer.length() > 40)
      {
        mpuBuffer = "";
      }
    }
  }
}

float normalizeError(float e)
{
  while (e > 180) e -= 360;
  while (e < -180) e += 360;
  return e;
}

// =====================================================
// LINE
// =====================================================
void updateLineSensor()
{
  if (!lineSensorEnabled)
  {
    lineState = 0;
    return;
  }

  raw2 = analogRead(OUT2_PIN);
  raw3 = analogRead(OUT3_PIN);
  raw4 = analogRead(OUT4_PIN);

  out2 = raw2 < LINE_THRESHOLD ? 0 : 1;
  out3 = raw3 < LINE_THRESHOLD ? 0 : 1;
  out4 = raw4 < LINE_THRESHOLD ? 0 : 1;

  lineState = 0;

  if (out2 == 0) lineState |= 0b100;
  if (out3 == 0) lineState |= 0b010;
  if (out4 == 0) lineState |= 0b001;
}

int readLineState()
{
  updateLineSensor();
  return lineState;
}

bool allSensorsZeroMarker()
{
  if (!lineSensorEnabled)
  {
    return false;
  }

  updateLineSensor();

  return out2 == 0 &&
         out3 == 0 &&
         out4 == 0;
}

// =====================================================
// DEBUG
// =====================================================
void printDebug()
{
  if (millis() - lastDebug >= debugInterval)
  {
    lastDebug = millis();

    Serial.print("MPU ");
    Serial.print("X:");
    Serial.print(mpuX);
    Serial.print(" Y:");
    Serial.print(mpuY);
    Serial.print(" Z:");
    Serial.print(mpuZ);

    if (lineSensorEnabled)
    {
      updateLineSensor();

      Serial.print(" || RAW ");
      Serial.print("OUT2/A1:");
      Serial.print(raw2);
      Serial.print(" OUT3/A2:");
      Serial.print(raw3);
      Serial.print(" OUT4/A3:");
      Serial.print(raw4);

      Serial.print(" || OUT ");
      Serial.print("OUT2:");
      Serial.print(out2);
      Serial.print(" OUT3:");
      Serial.print(out3);
      Serial.print(" OUT4:");
      Serial.print(out4);

      Serial.print(" || STATE:");
      Serial.print(lineState, BIN);
    }
    else
    {
      Serial.print(" || LINE:OFF");
    }

    Serial.print(" || ROBOT:");
    Serial.print(robotBusy ? "Unavailable" : "Available");

    Serial.print(" || WAIT_RETURN:");
    Serial.println(waitingReturn ? "YES" : "NO");
  }
}

void realtimeUpdate()
{
  readMPU();

  if (lineSensorEnabled)
  {
    updateLineSensor();
  }

  printDebug();
}

void waitMPUStableNoLine()
{
  unsigned long t = millis();

  while (millis() - t < 500)
  {
    readMPU();
    printDebug();
    delay(5);
  }
}

// =====================================================
// FOLLOW LINE
// =====================================================
void followLineStep()
{
  if (!lineSensorEnabled)
  {
    motorStop();
    return;
  }

  int state = readLineState();

  int leftSpeed = 0;
  int rightSpeed = 0;

  switch (state)
  {
    case 0b010:
      leftSpeed = BASE_SPEED;
      rightSpeed = BASE_SPEED;
      break;

    case 0b110:
    case 0b100:
      leftSpeed = 0;
      rightSpeed = LINE_TURN_SPEED;
      break;

    case 0b011:
    case 0b001:
      leftSpeed = LINE_TURN_SPEED;
      rightSpeed = 0;
      break;

    case 0b111:
      leftSpeed = BASE_SPEED;
      rightSpeed = BASE_SPEED;
      break;

    case 0b000:
      leftSpeed = BASE_SPEED;
      rightSpeed = BASE_SPEED;
      break;

    default:
      leftSpeed = BASE_SPEED;
      rightSpeed = BASE_SPEED;
      break;
  }

  leftSpeed *= LINE_DIRECTION_SIGN;
  rightSpeed *= LINE_DIRECTION_SIGN;

  driveWheels(leftSpeed, rightSpeed);
}

// =====================================================
// DI THANG QUA 3 MOC 000
// =====================================================
void goStraightThrough3Markers()
{
  forceLineOn();

  int markerCount = 0;
  bool lastMarker = false;

  unsigned long startEscape = millis();

  while (allSensorsZeroMarker() && millis() - startEscape < 800)
  {
    realtimeUpdate();
    followLineStep();
    delay(5);
  }

  while (markerCount < 3)
  {
    realtimeUpdate();

    bool markerNow = allSensorsZeroMarker();

    if (markerNow && !lastMarker)
    {
      markerCount++;

      Serial.print("MARKER COUNT:");
      Serial.println(markerCount);

      motorStop();
      delay(60);
    }

    lastMarker = markerNow;

    followLineStep();

    delay(5);
  }

  motorStop();
  delay(120);
}

// =====================================================
// SAU LENH CUOI:
// DI THEM DEN 1 MOC 000 ROI DUNG
// SAU DO TAT LINE VA RESET GOC VE 0
// =====================================================
void goStraightUntilOneMarker()
{
  forceLineOn();

  Serial.println("FINAL EXTRA STRAIGHT - WAIT 1 MARKER 000");

  bool lastMarker = false;

  unsigned long startEscape = millis();

  while (allSensorsZeroMarker() && millis() - startEscape < 1000)
  {
    realtimeUpdate();
    followLineStep();
    delay(5);
  }

  while (true)
  {
    realtimeUpdate();

    bool markerNow = allSensorsZeroMarker();

    if (markerNow && !lastMarker)
    {
      Serial.println("FINAL MARKER 000 DETECTED - STOP");

      motorStop();
      delay(300);

      // Tắt line ngay khi đến điểm cuối
      forceLineOff();

      // Reset góc về 0 bằng cách lấy yaw hiện tại làm mốc quay về
      readMPU();
      delay(50);
      readMPU();

      returnZeroYaw = currentYaw;

      Serial.print("RETURN ANGLE RESET TO 0 AT YAW:");
      Serial.println(returnZeroYaw);

      break;
    }

    lastMarker = markerNow;

    followLineStep();

    delay(5);
  }

  motorStop();
  delay(200);
}

// =====================================================
// PID QUAY 90 DO - KHONG DOC LINE
// Dung cho re trai/phai trong route
// =====================================================
void turnByAnglePIDNoLine(int dir, float degree)
{
  motorStop();
  delay(200);

  forceLineOff();

  waitMPUStableNoLine();

  float startYaw = currentYaw;
  float targetRelative = dir * (degree + TURN_EXTRA_DEGREE);

  float lastErr = targetRelative;
  unsigned long lastTime = millis();
  unsigned long turnStartTime = millis();

  Serial.print("TURN START PID NO LINE | Start Z:");
  Serial.print(startYaw);
  Serial.print(" Target:");
  Serial.println(targetRelative);

  while (true)
  {
    readMPU();
    printDebug();

    float turned = normalizeError(currentYaw - startYaw);
    float err = normalizeError(targetRelative - turned);

    if (abs(err) <= TURN_TOLERANCE)
    {
      break;
    }

    unsigned long now = millis();
    float dt = (now - lastTime) / 1000.0;
    if (dt <= 0) dt = 0.01;

    lastTime = now;

    float dErr = (err - lastErr) / dt;
    lastErr = err;

    float pidOutput = TURN_KP * err + TURN_KD * dErr;

    int wheelSpeed = abs(pidOutput);

    if (wheelSpeed > TURN_MAX_WHEEL_SPEED) wheelSpeed = TURN_MAX_WHEEL_SPEED;
    if (wheelSpeed < TURN_MIN_WHEEL_SPEED) wheelSpeed = TURN_MIN_WHEEL_SPEED;

    int turnDir = err > 0 ? 1 : -1;

    rotateInPlace(turnDir, wheelSpeed);

    Serial.print("TURN PID 90 | Turned:");
    Serial.print(turned);
    Serial.print(" Err:");
    Serial.print(err);
    Serial.print(" WheelSpeed:");
    Serial.print(wheelSpeed);
    Serial.print(" || LINE:");
    Serial.println(lineSensorEnabled ? "ON" : "OFF");

    if (millis() - turnStartTime > 8000)
    {
      Serial.println("TURN 90 TIMEOUT");
      break;
    }

    delay(10);
  }

  motorStop();
  delay(200);

  Serial.print("TURN 90 DONE | Current Z:");
  Serial.println(currentYaw);
}

// =====================================================
// QUAY TRAI 180 DO - KHONG LAC TRAI/PHAI
//
// Quan trong:
// - Line OFF toan bo qua trinh quay
// - Lay returnZeroYaw lam goc 0
// - PID chi dieu khien toc do
// - Chieu quay luon la TRAI
// =====================================================
void turnBack180LeftPIDNoLine()
{
  Serial.println("TURN BACK LEFT 180 PID - LINE OFF");

  motorStop();
  delay(300);

  forceLineOff();
  delay(300);

  waitMPUStableNoLine();

  // Reset goc ve 0 ngay truoc khi quay
  returnZeroYaw = currentYaw;

  float lastYaw = returnZeroYaw;
  float turnedAbs = 0;

  float lastErr = TURN_BACK_TARGET;
  unsigned long lastTime = millis();
  unsigned long turnStartTime = millis();

  Serial.print("ANGLE ZERO BEFORE LEFT 180 | YAW:");
  Serial.println(returnZeroYaw);

  while (true)
  {
    // Chi doc MPU, khong doc line
    readMPU();
    printDebug();

    // Cong don goc quay theo tung buoc de tranh loi 179 -> -180
    float delta = normalizeError(currentYaw - lastYaw);
    lastYaw = currentYaw;

    turnedAbs += abs(delta);

    float err = TURN_BACK_TARGET - turnedAbs;

    if (err <= TURN_TOLERANCE)
    {
      break;
    }

    unsigned long now = millis();
    float dt = (now - lastTime) / 1000.0;
    if (dt <= 0) dt = 0.01;

    lastTime = now;

    float dErr = (err - lastErr) / dt;
    lastErr = err;

    float pidOutput = TURN_KP * err + TURN_KD * dErr;

    int wheelSpeed = abs(pidOutput);

    if (wheelSpeed > TURN_MAX_WHEEL_SPEED) wheelSpeed = TURN_MAX_WHEEL_SPEED;
    if (wheelSpeed < TURN_MIN_WHEEL_SPEED) wheelSpeed = TURN_MIN_WHEEL_SPEED;

    // QUAN TRONG:
    // Chi quay trai, khong cho PID doi chieu sang phai
    rotateInPlace(-1, wheelSpeed);

    Serial.print("TURN LEFT 180 PID | Now:");
    Serial.print(currentYaw);
    Serial.print(" Delta:");
    Serial.print(delta);
    Serial.print(" TurnedAbs:");
    Serial.print(turnedAbs);
    Serial.print(" Err:");
    Serial.print(err);
    Serial.print(" WheelSpeed:");
    Serial.print(wheelSpeed);
    Serial.print(" || LINE:");
    Serial.println(lineSensorEnabled ? "ON" : "OFF");

    if (millis() - turnStartTime > 9000)
    {
      Serial.println("TURN LEFT 180 TIMEOUT");
      break;
    }

    delay(10);
  }

  motorStop();
  delay(300);

  Serial.println("LEFT 180 TURN DONE - LINE STILL OFF");
}

// =====================================================
// TURN 90
// =====================================================
void turnRight90()
{
  turnByAnglePIDNoLine(1, 90);
  forceLineOn();
}

void turnLeft90()
{
  turnByAnglePIDNoLine(-1, 90);
  forceLineOn();
}

// =====================================================
// EXECUTE 1 COMMAND
// =====================================================
void executeOneCommand(char cmd)
{
  Serial.print("EXECUTE CMD:");
  Serial.println(cmd);

  forceLineOn();

  goStraightThrough3Markers();

  if (cmd == 'R')
  {
    Serial.println("START TURN RIGHT 90");
    turnRight90();
  }
  else if (cmd == 'L')
  {
    Serial.println("START TURN LEFT 90");
    turnLeft90();
  }
  else if (cmd == 'T')
  {
    Serial.println("GO STRAIGHT ONLY");
  }
}

// =====================================================
// BUILD RETURN ROUTE
// VD: aThLn -> aRhTn
// =====================================================
void buildReturnRoute()
{
  returnLen = 0;

  for (int i = routeLen - 1; i >= 0; i--)
  {
    char c = routeCmd[i];

    if (c == 'R') c = 'L';
    else if (c == 'L') c = 'R';
    else c = 'T';

    if (returnLen < MAX_CMD)
    {
      returnCmd[returnLen] = c;
      returnLen++;
    }
  }
}

// =====================================================
// PARSE COMMAND
// =====================================================
void parseCommandString(String s)
{
  routeLen = 0;
  returnLen = 0;
  hasRoute = false;
  waitingReturn = false;

  if (s.length() < 3) return;
  if (s[0] != 'a') return;
  if (s[s.length() - 1] != 'n') return;

  for (int i = 1; i < s.length() - 1; i++)
  {
    char c = s[i];

    if (c == 'h') continue;

    if (c == 'T' || c == 'R' || c == 'L')
    {
      if (routeLen < MAX_CMD)
      {
        routeCmd[routeLen] = c;
        routeLen++;
      }
    }
  }

  if (routeLen > 0)
  {
    hasRoute = true;
    waitingReturn = false;

    buildReturnRoute();

    printRouteString("ROUTE STRING", routeCmd, routeLen);
    printRouteString("RETURN STRING", returnCmd, returnLen);
  }
}

// =====================================================
// EXECUTE ROUTE DI
// =====================================================
void executeRoute(char *cmdList, int len)
{
  robotBusy = true;
  waitingReturn = false;

  sendStatus("Unavailable");

  for (int i = 0; i < len; i++)
  {
    executeOneCommand(cmdList[i]);
  }

  // Chay het lenh cuoi, di thang them den 1 moc 000 roi dung
  goStraightUntilOneMarker();

  motorStop();

  sendStatus("ok");
  sendStatus("Available");

  robotBusy = false;
  waitingReturn = true;

  Serial.println("WAIT BUTTON TO RETURN");
  printRouteString("RETURN WILL RUN", returnCmd, returnLen);
}

// =====================================================
// READ COMMAND UART
// =====================================================
void readCommandUART()
{
  while (Serial.available())
  {
    char c = Serial.read();

    if (c == 'a')
    {
      receivingCommand = true;
      inputBuffer = "";
      inputBuffer += c;
    }
    else if (receivingCommand)
    {
      inputBuffer += c;

      if (c == 'n')
      {
        receivingCommand = false;

        Serial.print("RECEIVED COMMAND STRING:");
        Serial.println(inputBuffer);

        if (robotBusy || waitingReturn)
        {
          sendStatus("Unavailable");
          Serial.println("WAITING BUTTON RETURN - NEW COMMAND IGNORED");
          inputBuffer = "";
        }
        else
        {
          parseCommandString(inputBuffer);

          if (hasRoute)
          {
            executeRoute(routeCmd, routeLen);
          }
          else
          {
            Serial.println("COMMAND ERROR");
            resetAllCommandData();
          }
        }
      }
    }
  }
}

// =====================================================
// BUTTON D2
// =====================================================
void checkButton()
{
  static bool lastBtn = HIGH;

  bool btn = digitalRead(BUTTON_PIN);

  if (lastBtn == HIGH && btn == LOW)
  {
    delay(50);

    if (digitalRead(BUTTON_PIN) == LOW)
    {
      Serial.println("BUTTON PRESSED");

      if (!robotBusy && waitingReturn && hasRoute && returnLen > 0)
      {
        sendStatus("received");

        // Luu lenh nguoc ra bien tam
        char tempReturnCmd[MAX_CMD];
        int tempReturnLen = returnLen;

        for (int i = 0; i < MAX_CMD; i++)
        {
          tempReturnCmd[i] = 0;
        }

        for (int i = 0; i < tempReturnLen; i++)
        {
          tempReturnCmd[i] = returnCmd[i];
        }

        printRouteString("TEMP RETURN SAVED", tempReturnCmd, tempReturnLen);

        robotBusy = true;
        waitingReturn = false;

        sendStatus("Unavailable");

        // Dung xe
        motorStop();
        delay(300);

        // Reset runtime nhung khong xoa lenh tam
        resetRuntimeBeforeReturn();
        delay(200);

        // Tat line truoc khi quay
        forceLineOff();
        delay(300);

        // Quay trai 180 do, PID chi dieu khien toc do
        turnBack180LeftPIDNoLine();

        // Quay xong moi bat line
        forceLineOn();
        delay(500);

        Serial.println("LINE ON - START RETURN ROUTE");

        printRouteString("RUN RETURN", tempReturnCmd, tempReturnLen);

        // Chay nguoc lai lenh ban dau
        for (int i = 0; i < tempReturnLen; i++)
        {
          executeOneCommand(tempReturnCmd[i]);
        }

        motorStop();

        sendStatus("ok");
        sendStatus("Available");

        robotBusy = false;

        resetAllCommandData();
      }
      else
      {
        Serial.print("BUTTON IGNORED | robotBusy=");
        Serial.print(robotBusy);
        Serial.print(" waitingReturn=");
        Serial.print(waitingReturn);
        Serial.print(" hasRoute=");
        Serial.print(hasRoute);
        Serial.print(" returnLen=");
        Serial.println(returnLen);
      }
    }
  }

  lastBtn = btn;
}

// =====================================================
// SETUP
// =====================================================
void setup()
{
  Serial.begin(9600);

  MpuSerial.begin(9600);
  HoverSerial.begin(HOVER_BAUD);

  pinMode(BUTTON_PIN, INPUT_PULLUP);

  pinMode(OUT2_PIN, INPUT);
  pinMode(OUT3_PIN, INPUT);
  pinMode(OUT4_PIN, INPUT);

  delay(1000);

  MpuSerial.listen();

  resetAllCommandData();

  sendStatus("UNO READY");
  sendStatus("Available");
}

// =====================================================
// LOOP
// =====================================================
